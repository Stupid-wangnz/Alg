简单使用`find(nums.begin() + i + 1, nums.end(), target - nums[i]);distance(nums.begin(), it); `

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int>res;
        int n = nums.size();
        for(int i=0; i<n; i++){
            auto it = find(nums.begin() + i + 1, nums.end(), target - nums[i]);
            if(it != nums.end()){
                int index = distance(nums.begin(), it);
                vector<int>res = {i, index};
                return res;
            }
        }
        return res;
    }
};
```

**map, faster**

```c++
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int n = nums.size();
        map<int, int>m_num;
        vector<int>res;
        for(int i = 0; i<n; i++){
            if(m_num.count(target - nums[i]) != 0){
                res = {m_num[target - nums[i]], i};
                return res;
            }
            m_num[nums[i]] = i;
        }
        return res;
    }
};
```
