O(n^2)

```
class Solution {
public:
    int maxArea(vector<int>& height) {
        int n =height.size();
        int maxarea = 0;
        for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                maxarea = max(min(height[i], height[j]) * (j - i), maxarea);
            }
        }
        return maxarea;
    }
};
```



O(n):双指针

```
class Solution {
public:
    int maxArea(vector<int>& height) {
        int r =height.size() -1, l = 0;
        int maxarea = 0;
        while(l<r){
            maxarea = max(maxarea, (r - l) * min(height[r], height[l]));

            if(height[r] < height[l])
                r--;
            else
                l++;
        }
        return maxarea;
    }
};
```
